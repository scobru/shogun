SHOGUN - Privacy-First Wallet Manager

A decentralized wallet manager using Gun.js to handle wallets and private keys directly in the browser. Featuring a complete authentication system with stealth address support and end-to-end encryption.

How It Works

SHOGUN uses Gun.js as a foundation to create a fully decentralized wallet management system. Private keys are managed locally, while synchronization and storage happen in a distributed manner. Stealth addresses provide an additional layer of privacy for transactions.

All critical operations happen in the user's browser. Sensitive data never leaves the user's device and is always encrypted end-to-end.

Quick Start

// Initialize with default Gun configuration
import { WalletManager, StorageType } from '@scobru/shogun';

const manager = new WalletManager();

// Create account
try {
  await manager.createAccount('username', 'password');
  console.log('Account created successfully!');
} catch (error) {
  console.error('Error creating account:', error);
}

// Login
try {
  const pubKey = await manager.login('username', 'password');
  console.log('Logged in with public key:', pubKey);
} catch (error) {
  console.error('Error logging in:', error);
}

// 1. Create a new wallet from Gun key pair
const gunKeyPair = manager.getCurrentUserKeyPair();
const { walletObj, entropy } = await WalletManager.createWalletObj(gunKeyPair);
console.log('New wallet created:', walletObj);
console.log('Entropy (save this!):', entropy);

// 2. Create a wallet from existing salt/entropy
const existingWallet = await WalletManager.createWalletFromSalt(gunKeyPair, entropy);
console.log('Wallet recreated from entropy:', existingWallet);

// 3. Save wallet (both to Gun and localStorage)
await manager.saveWallet(walletObj, pubKey, StorageType.BOTH);

// 4. Retrieve wallet
const retrievedWallet = await manager.retrieveWallet(pubKey, StorageType.BOTH);
console.log('Retrieved wallet:', retrievedWallet);

// 5. Export wallet data
const exportedData = await manager.exportAllData('username');
console.log('Exported wallet data:', exportedData);

// 6. Import wallet data
await manager.importAllData(exportedData, 'username');

// 7. Check wallet status
const status = await manager.checkLocalData('username');
console.log('Wallet status:', status);

Testing Examples

// Test wallet creation and management
describe("Wallet Management", function() {
  let walletManager;
  let testAlias;

  beforeEach(async () => {
    walletManager = new WalletManager();
    testAlias = generateRandomAlias();
    await walletManager.createAccount(testAlias, "password123");
  });

  it("should save and retrieve wallet", async () => {
    const testWallet = new Wallet("0xTestPublicKey");
    await walletManager.saveWalletLocally(testWallet, testAlias);
    const retrieved = await walletManager.retrieveWalletLocally(testAlias);
    assert.strictEqual(retrieved.publicKey, testWallet.publicKey);
  });

  it("should export and import data", async () => {
    const exported = await walletManager.exportAllData(testAlias);
    await walletManager.clearLocalData(testAlias);
    await walletManager.importAllData(exported, testAlias);
    const status = await walletManager.checkLocalData(testAlias);
    assert(status.hasWallet);
  });
});

Advanced Examples

// Complete data export/import example
async function backupAndRestore() {
  const manager = new WalletManager();
  
  // Login and setup
  await manager.login('username', 'password');
  
  // Create and save wallet
  const gunKeyPair = manager.getCurrentUserKeyPair();
  const { walletObj } = await WalletManager.createWalletObj(gunKeyPair);
  await manager.saveWallet(walletObj, manager.getPublicKey());
  
  // Export all data
  const backup = await manager.exportAllData('username');
  
  // Clear local data
  await manager.clearLocalData('username');
  
  // Import data back
  await manager.importAllData(backup, 'username');
  
  // Verify data integrity
  const status = await manager.checkLocalData('username');
  console.log('Data restored:', status);
}
